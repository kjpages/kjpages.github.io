def _hart6(x,
           alpha=np.asarray([1.0, 1.2, 3.0, 3.2]),
           P=10**-4 * np.asarray([[1312, 1696, 5569, 124, 8283, 5886],
                                  [2329, 4135, 8307, 3736, 1004, 9991],
                                  [2348, 1451, 3522, 2883, 3047, 6650],
                                  [4047, 8828, 8732, 5743, 1091, 381]]),
           A=np.asarray([[10, 3, 17, 3.50, 1.7, 8],
                         [0.05, 10, 17, 0.1, 8, 14],
                         [3, 3.5, 1.7, 10, 17, 8],
                         [17, 8, 0.05, 10, 0.1, 14]])):
    """The six dimensional Hartmann function is defined on the unit hypercube.
    It has six local minima and one global minimum f(x*) = -3.32237 at
    x* = (0.20169, 0.15001, 0.476874, 0.275332, 0.311652, 0.6573).
    More details: <http://www.sfu.ca/~ssurjano/hart6.html>
    """
    return -np.sum(alpha * np.exp(-np.sum(A * (np.array(x) - P)**2, axis=1)))
# build noisy observations

def hart6(x, noise_level=0.):
    return _hart6(x) + noise_level * np.random.randn()

from skopt import gp_minimize, forest_minimize, dummy_minimize
from functools import partial

f_hart6 = partial(hart6, noise_level=0.01)
bounds = np.tile((0., 1.), (6, 1))
n_calls = 100

def run_hart6_skopt(minimizer, n_iter=5):
    results = []
    for n in range(n_iter):
        result = minimizer(f_hart6, bounds, n_calls=n_calls, random_state=n)
        results.append(result)
    return results

# Random search
dummy_hart6 = run_hart6_skopt(dummy_minimize)

# hyperopt: TPE
# results not very stable
# need to define separate recorder for points raised 

from hyperopt import fmin, tpe, hp

Xi = []
yi = []

def hart6_wrapped(x, noise_level=0.01):
    Xi.append(x)
    y = _hart6(x) + noise_level * np.random.randn()
    yi.append(y)
    return y

hyperopt_model_hart6 = fmin(
    fn=hart6_wrapped,
    space=hp.choice('x', [[hp.uniform('x1', 0., 1.), hp.uniform('x2', 0., 1.),
                           hp.uniform('x3', 0., 1.), hp.uniform('x4', 0., 1.),
                           hp.uniform('x5', 0., 1.), hp.uniform('x6', 0., 1.)]]),
    algo=tpe.suggest,
    max_evals=100)

import GPyOpt

bounds = np.tile((0., 1.), (6, 1))
domain = [{'name': 'var_1', 'type': 'continuous', 'domain': bounds[0]}, ## use default bounds
          {'name': 'var_2', 'type': 'continuous', 'domain': bounds[1]},
          {'name': 'var_3', 'type': 'continuous', 'domain': bounds[2]},
          {'name': 'var_4', 'type': 'continuous', 'domain': bounds[3]},
          {'name': 'var_5', 'type': 'continuous', 'domain': bounds[4]},
          {'name': 'var_6', 'type': 'continuous', 'domain': bounds[5]}]
batch_size = 4
num_cores = 4
n_calls = 25

gpmcmc_minimize = GPyOpt.methods.BayesianOptimization(f=hart6,
                                                      domain = domain,
                                                      acquisition_type = 'EI',
                                                      normalize_Y = True,
                                                      initial_design_numdata = 10,
                                                      evaluator_type = 'local_penalization',
                                                      batch_size = batch_size,
                                                      num_cores = num_cores,
                                                      acquisition_jitter = 0)
gpmcmc_minimize.run_optimization(n_calls) # 254=100 final number of evaluations
